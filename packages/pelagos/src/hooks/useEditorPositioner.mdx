## Example

```js
import {useRef} from 'react';
import {useEditorPositioner} from '@bluecateng/pelagos';

const Example = () => {
	const editorRef = useRef(null);
	useEditorPositioner(editorRef, 'button', 'track');
	return <div ref={editorRef}>...</div>;
};
```

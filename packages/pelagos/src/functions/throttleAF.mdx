## Example

```js
import {useMemo} from 'react';
import {throttleAF} from '@bluecateng/pelagos';

const Example = () => {
	const handleChange = useMemo(
		() =>
			throttleAF(() => {
				// a task that may take some time
			}),
		[]
	);
	return <input onChange={handleChange} />;
};
```
